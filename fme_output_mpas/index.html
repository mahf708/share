 277  def find_files(rundir, pattern, exclude=None):
 278      """Find files matching glob pattern, optionally excluding a substring.
 279  
 280      Parameters
 281      ----------
 282      rundir : str
 283          Directory to search in.
 284      pattern : str
 285          Glob pattern relative to rundir.
 286      exclude : str or None
 287          If set, exclude any file whose basename contains this substring.
 288  
 289      Returns
 290      -------
 291      list of str
 292          Sorted list of matching file paths.
 293      """
 294      hits = sorted(glob.glob(os.path.join(rundir, pattern)))
 295      # Always exclude .base restart files
 296      hits = [f for f in hits if not f.endswith('.base')]
 297      if exclude:
 298          hits = [f for f in hits if exclude not in os.path.basename(f)]
 299      return hits

 353  def valid_data(arr, fill=1e10):
 354      """Return non-fill, finite values.
 355  
 356      The threshold is 1e10 because remapped output can contain interpolated
 357      fill artifacts from neighboring land/ocean cells that can reach ~1e14
 358      scale while _FillValue is 1e34.  No geophysical quantity exceeds 1e10.
 359      """
 360      if arr is None:
 361          return None
 362      flat = arr.ravel().astype(float)
 363      mask = (np.abs(flat) < fill) & np.isfinite(flat)
 364      return flat[mask] if mask.any() else None

 367  def fill_nan_report(arr, name):
 368      """Return a dict with fill/NaN fraction stats for a variable array."""
 369      if arr is None:
 370          return {"name": name, "total": 0, "valid": 0, "fill_nan": 0, "pct": 100.0}
 371      flat = arr.ravel().astype(float)
 372      total = flat.size
 373      fill_mask = np.abs(flat) >= 1e10
 374      nan_mask = ~np.isfinite(flat)
 375      bad = fill_mask | nan_mask
 376      n_bad = int(bad.sum())
 377      n_valid = total - n_bad
 378      pct = 100.0 * n_bad / total if total > 0 else 0.0
 379      return {"name": name, "total": total, "valid": n_valid,
 380              "fill_nan": n_bad, "pct": pct}

 426  def _extract_field(arr, mode):
 427      """Extract a 1D (nCells) field from a multi-dimensional array.
 428  
 429      mode: 'last2d' = last timestep of (Time, nCells)
 430            'surface' = last timestep, level 0 of (Time, nVertLevels, nCells)
 431      """

 433          return None
 434      if mode == "surface":
 435          if arr.ndim == 3:      # (Time, nVertLevels, nCells)
 436              return arr[-1, 0, :]

 438              return arr[0, :]

 440          if arr.ndim == 2:      # (Time, nCells)
 441              return arr[-1, :]
 442          elif arr.ndim == 1:    # (nCells)
 443              return arr[:]
 444      return None

 447  def _diff_stats(leg, fme, fill_thresh=1e10):
 448      """Compute difference statistics between two 1D fields on the same grid.
 449  
 450      Returns dict with n_valid, legacy_mean, fme_mean, diff_mean, diff_rms,
 451      diff_max_abs, correlation -- or None if no valid overlap.
 452      """
 453      if leg is None or fme is None:
 454          return None
 455      a = leg.ravel().astype(float)
 456      b = fme.ravel().astype(float)
 457      if a.size != b.size:
 458          return None
 459      ok = ((np.abs(a) < fill_thresh) & np.isfinite(a) &
 460            (np.abs(b) < fill_thresh) & np.isfinite(b))
 461      if ok.sum() == 0:
 462          return None
 463      a, b = a[ok], b[ok]
 464      d = b - a
 465      s = {
 466          "n_valid": int(ok.sum()),
 467          "legacy_mean": float(a.mean()),
 468          "fme_mean": float(b.mean()),
 469          "diff_mean": float(d.mean()),
 470          "diff_rms": float(np.sqrt((d ** 2).mean())),
 471          "diff_max_abs": float(np.abs(d).max()),
 472      }
 473      if a.std() > 0 and b.std() > 0:
 474          s["correlation"] = float(np.corrcoef(a, b)[0, 1])
 475      else:
 476          s["correlation"] = float("nan")
 477      return s

 502  def _plot_on_ax(ax, data, lons, lats, cmap, vmin, vmax, is_cartopy):
 503      """Plot data on an axis -- tripcolor for 1D, pcolormesh for 2D."""
 504      # Mask fill values so they render as transparent (not clipped to vmax)
 505      data = np.where((np.abs(data) < 1e10) & np.isfinite(data),
 506                      data.astype(float), np.nan)
 507      transform = ccrs.PlateCarree() if is_cartopy else None
 508      if data.ndim == 2:
 509          kw = dict(transform=transform) if is_cartopy else {}
 510          im = ax.pcolormesh(lons, lats, data, cmap=cmap,
 511                             vmin=vmin, vmax=vmax, shading="auto", **kw)
 512      else:
 513          # Unstructured: use tripcolor for filled triangulation.
 514          # Subsample if > 50k points to keep triangulation fast.
 515          # Mask triangles that cross the antimeridian (lon span > 180).
 516          from matplotlib.tri import Triangulation
 517          lons_fix = _fix_lon(lons)
 518          kw = dict(transform=ccrs.PlateCarree()) if is_cartopy else {}
 519          max_pts = 50000
 520          if data.size > max_pts:
 521              idx = np.random.default_rng(42).choice(data.size, max_pts, replace=False)
 522              lons_sub, lats_sub, data_sub = lons_fix[idx], lats[idx], data[idx]
 523          else:
 524              lons_sub, lats_sub, data_sub = lons_fix, lats, data
 525          try:
 526              tri = Triangulation(lons_sub, lats_sub)
 527              # Mask triangles spanning the antimeridian
 528              tri_lons = lons_sub[tri.triangles]
 529              lon_span = tri_lons.max(axis=1) - tri_lons.min(axis=1)
 530              tri.set_mask(lon_span > 180)
 531              im = ax.tripcolor(tri, data_sub, cmap=cmap,
 532                                vmin=vmin, vmax=vmax, **kw)
 533          except Exception:
 534              # Fallback to scatter if tripcolor fails
 535              im = ax.scatter(lons_sub, lats_sub, c=data_sub, s=0.5, cmap=cmap,
 536                              vmin=vmin, vmax=vmax, **kw)
 537      return im

 735  def depth_latitude_section(ds, base_varname, n_levels, title, outdir, fname,
 736                             cmap="RdBu_r", vmin=None, vmax=None, units=""):
 737      """Plot a depth-latitude cross-section from per-level remapped variables.
 738  
 739      For each level k (0 to n_levels-1) reads {base_varname}_{k}, extracts the
 740      last timestep, computes zonal mean, and stacks into a (n_levels, nlat) array.
 741      Depth midpoints are computed from DEPTH_BOUNDS.
 742      """
 743      if not HAS_MPL:
 744          return None
 745  
 746      lat = get_var(ds, "lat")
 747      if lat is None:
 748          return None
 749  
 750      data = []
 751      for k in range(n_levels):
 752          vname = f"{base_varname}_{k}"
 753          arr = get_var(ds, vname)
 754          zm = _compute_zonal_mean(arr)
 755          if zm is None:
 756              return None
 757          data.append(zm)
 758  
 759      data = np.array(data)  # (n_levels, nlat)
 760      n_bounds = min(n_levels + 1, len(DEPTH_BOUNDS))
 761      depth_mid = np.array([(DEPTH_BOUNDS[k] + DEPTH_BOUNDS[k + 1]) / 2.0
 762                            for k in range(n_bounds - 1)])
 763      # Trim data to match available depth bounds
 764      n_plot = min(len(depth_mid), data.shape[0])
 765      depth_mid = depth_mid[:n_plot]
 766      data = data[:n_plot, :]
 767  
 768      fig, ax = plt.subplots(figsize=(10, 5))
 769      im = ax.pcolormesh(lat, depth_mid, data, cmap=cmap,
 770                         vmin=vmin, vmax=vmax, shading="auto")

1091      """Verify that a remapped dataset has the expected lat-lon grid.

1114  
1115      if not issues:
1116          nlon = dims.get("lon", "?")
1117          nlat = dims.get("lat", "?")

1703  def _area_weighted_mean(data_1d, area_1d, fill_thresh=1e10):
1704      """Area-weighted mean of a 1D field, excluding fill values."""
1705      ok = (np.abs(data_1d) < fill_thresh) & np.isfinite(data_1d) & np.isfinite(area_1d)
1706      if ok.sum() == 0:
1707          return float("nan")
1708      return float(np.sum(data_1d[ok] * area_1d[ok]) / np.sum(area_1d[ok]))
1709  
1710  
1711  def _cosine_weighted_mean(data_2d, lat_1d, fill_thresh=1e10):
1712      """Cosine-of-latitude-weighted mean of a (lat, lon) field."""
1713      w = np.cos(np.deg2rad(lat_1d))
1714      w2d = np.broadcast_to(w[:, None], data_2d.shape)
1715      ok = (np.abs(data_2d) < fill_thresh) & np.isfinite(data_2d)
1716      if ok.sum() == 0:
1717          return float("nan")
1718      return float(np.sum(data_2d[ok] * w2d[ok]) / np.sum(w2d[ok]))

1725  def check_remap_conservation(rundir, outdir, verbose):
1726      """Verify area-weighted global means match between native and remapped grids."""
1727      print("\n=== Remap Conservation Check ===")
1728      issues = []
1729      plots = []
1730      fill_reports = []
1731  
1732      nat_files = find_files(rundir, "*.mpaso.hist.am.fmeDerivedFields.*.nc",
1733                             exclude=".remapped.")
1734      rem_files = find_files(rundir, "*.mpaso.hist.am.fmeDerivedFields.*.remapped.nc")
1735      if not nat_files or not rem_files:
1736          print("  SKIP: need both native and remapped fmeDerivedFields files")
1737          return issues, plots, fill_reports
1738  
1739      ds_nat = safe_open(nat_files[0])
1740      ds_rem = safe_open(rem_files[0])
1741      if has_time_zero(ds_nat) or has_time_zero(ds_rem):
1742          print("  SKIP: time dimension empty")
1743          close_ds(ds_nat); close_ds(ds_rem)
1744          return issues, plots, fill_reports
1745  
1746      area = get_var(ds_nat, "areaCell")  # may be None in AM output
1747      lat = get_var(ds_rem, "lat")
1748      if lat is None:
1749          print("  SKIP: missing lat in remapped file")
1750          close_ds(ds_nat); close_ds(ds_rem)
1751          return issues, plots, fill_reports
1752  
1753      tol = 0.05  # 5% relative tolerance (remap + fill boundary effects)
1754      for vname in ["sst", "ssh"]:
1755          arr_nat = get_var(ds_nat, vname)
1756          arr_rem = get_var(ds_rem, vname)
1757          if arr_nat is None or arr_rem is None:
1758              continue
1759          d_nat = arr_nat[-1] if arr_nat.ndim > 1 else arr_nat
1760          d_rem = arr_rem[-1] if arr_rem.ndim == 3 else arr_rem
1761  
1762          # Native: area-weighted if areaCell available, else unweighted
1763          # (MPAS quasi-uniform mesh: unweighted is a good approximation)
1764          if area is not None:
1765              m_nat = _area_weighted_mean(d_nat, area)
1766          else:
1767              v = valid_data(d_nat)
1768              m_nat = float(v.mean()) if v is not None and v.size > 0 else float("nan")
1769          m_rem = _cosine_weighted_mean(d_rem, lat)
1770          denom = max(abs(m_nat), 1e-30)
1771          rel_diff = abs(m_nat - m_rem) / denom
1772          status = "PASS" if rel_diff < tol else "FAIL"
1773          if status == "FAIL":
1774              issues.append(f"  {vname}: native={m_nat:.4g} remap={m_rem:.4g} "
1775                            f"rel_diff={rel_diff:.2%}")
1776          print(f"  {vname:6s}  native={m_nat:+.4f}  remap={m_rem:+.4f}  "
1777                f"rel_diff={rel_diff:.2%}  {status}")
1778  
1779      close_ds(ds_nat); close_ds(ds_rem)
1780      _report("Remap conservation", issues)
1781      return issues, plots, fill_reports
