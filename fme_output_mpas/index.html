 134  def find_files(rundir, pattern, exclude=None):
 135      """Find files matching glob pattern, optionally excluding a substring.
 136  
 137      Parameters
 138      ----------
 139      rundir : str
 140          Directory to search in.
 141      pattern : str
 142          Glob pattern relative to rundir.
 143      exclude : str or None
 144          If set, exclude any file whose basename contains this substring.
 145  
 146      Returns
 147      -------
 148      list of str
 149          Sorted list of matching file paths.
 150      """
 151      hits = sorted(glob.glob(os.path.join(rundir, pattern)))
 152      # Always exclude .base restart files
 153      hits = [f for f in hits if not f.endswith('.base')]
 154      if exclude:
 155          hits = [f for f in hits if exclude not in os.path.basename(f)]
 156      return hits

 210  def valid_data(arr, fill=1e10):
 211      """Return non-fill, finite values.
 212  
 213      The threshold is 1e10 because remapped output can contain interpolated
 214      fill artifacts from neighboring land/ocean cells that can reach ~1e14
 215      scale while _FillValue is 1e34.  No geophysical quantity exceeds 1e10.
 216      """
 217      if arr is None:
 218          return None
 219      flat = arr.ravel().astype(float)
 220      mask = (np.abs(flat) < fill) & np.isfinite(flat)
 221      return flat[mask] if mask.any() else None

 224  def fill_nan_report(arr, name):
 225      """Return a dict with fill/NaN fraction stats for a variable array."""
 226      if arr is None:
 227          return {"name": name, "total": 0, "valid": 0, "fill_nan": 0, "pct": 100.0}
 228      flat = arr.ravel().astype(float)
 229      total = flat.size
 230      fill_mask = np.abs(flat) >= 1e10
 231      nan_mask = ~np.isfinite(flat)
 232      bad = fill_mask | nan_mask
 233      n_bad = int(bad.sum())
 234      n_valid = total - n_bad
 235      pct = 100.0 * n_bad / total if total > 0 else 0.0
 236      return {"name": name, "total": total, "valid": n_valid,
 237              "fill_nan": n_bad, "pct": pct}

 240  def check_range(arr, name, vmin=None, vmax=None):
 241      v = valid_data(arr)
 242      issues = []
 243      if v is None or v.size == 0:
 244          issues.append(f"  {name}: no valid data")
 245          return issues
 246      if not np.isfinite(v).all():
 247          issues.append(f"  {name}: NaN/Inf present")
 248      if vmin is not None and v.min() < vmin:
 249          issues.append(f"  {name}: min={v.min():.4g} below {vmin}")
 250      if vmax is not None and v.max() > vmax:
 251          issues.append(f"  {name}: max={v.max():.4g} above {vmax}")
 252      return issues

 305  def _plot_on_ax(ax, data, lons, lats, cmap, vmin, vmax, is_cartopy):
 306      """Plot data on an axis -- tripcolor for 1D, pcolormesh for 2D."""
 307      transform = ccrs.PlateCarree() if is_cartopy else None
 308      if data.ndim == 2:
 309          kw = dict(transform=transform) if is_cartopy else {}
 310          im = ax.pcolormesh(lons, lats, data, cmap=cmap,
 311                             vmin=vmin, vmax=vmax, shading="auto", **kw)
 312      else:
 313          # Unstructured: use tripcolor for filled triangulation.
 314          # Subsample if > 50k points to keep triangulation fast.
 315          # Mask triangles that cross the antimeridian (lon span > 180).
 316          from matplotlib.tri import Triangulation
 317          lons_fix = _fix_lon(lons)
 318          kw = dict(transform=ccrs.PlateCarree()) if is_cartopy else {}
 319          max_pts = 50000
 320          if data.size > max_pts:
 321              idx = np.random.default_rng(42).choice(data.size, max_pts, replace=False)
 322              lons_sub, lats_sub, data_sub = lons_fix[idx], lats[idx], data[idx]
 323          else:
 324              lons_sub, lats_sub, data_sub = lons_fix, lats, data
 325          try:
 326              tri = Triangulation(lons_sub, lats_sub)
 327              # Mask triangles spanning the antimeridian
 328              tri_lons = lons_sub[tri.triangles]
 329              lon_span = tri_lons.max(axis=1) - tri_lons.min(axis=1)
 330              tri.set_mask(lon_span > 180)
 331              im = ax.tripcolor(tri, data_sub, cmap=cmap,
 332                                vmin=vmin, vmax=vmax, **kw)
 333          except Exception:
 334              # Fallback to scatter if tripcolor fails
 335              im = ax.scatter(lons_sub, lats_sub, c=data_sub, s=0.5, cmap=cmap,
 336                              vmin=vmin, vmax=vmax, **kw)
 337      return im

 341                 outdir=None, fname=None, units="", subdir=None):
 342      """

 521  def _compute_zonal_mean(arr, fill_thresh=1e10):
 522      """Compute zonal mean from (lat, lon) array, masking fill values."""
 523      if arr is None:
 524          return None
 525      if arr.ndim == 3:
 526          data = arr[-1, :, :]
 527      elif arr.ndim == 2:
 528          data = arr
 529      else:
 530          return None
 531      masked = np.where(np.abs(data) < fill_thresh, data, np.nan)
 532      return np.nanmean(masked, axis=1)

 571  def check_remapped_grid(ds, label):
 572      """Verify that a remapped dataset has the expected lat-lon grid.
 573  
 574      Returns a list of issues (empty if OK).
 575      """
 576      issues = []
 577      dims = get_dims(ds)
 578  
 579      # Check for lon/lat dimensions
 580      has_lon = "lon" in dims
 581      has_lat = "lat" in dims
 582      if not has_lon:
 583          issues.append(f"  {label}: missing 'lon' dimension")
 584      if not has_lat:
 585          issues.append(f"  {label}: missing 'lat' dimension")
 586      if has_lon and dims["lon"] != EXPECTED_NLON:
 587          issues.append(f"  {label}: lon={dims['lon']}, expected {EXPECTED_NLON}")
 588      if has_lat and dims["lat"] != EXPECTED_NLAT:
 589          issues.append(f"  {label}: lat={dims['lat']}, expected {EXPECTED_NLAT}")
 590  
 591      # Check for Time dimension
 592      has_time = "Time" in dims or "time" in dims
 593      if not has_time:
 594          issues.append(f"  {label}: missing 'Time' dimension")
 595  
 596      if not issues:
 597          nlon = dims.get("lon", "?")
 598          nlat = dims.get("lat", "?")
 599          print(f"  {label}: lat-lon grid OK ({nlon}x{nlat})")
 600  
 601      return issues
