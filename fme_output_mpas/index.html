 341      nan_mask = ~np.isfinite(flat)
 342      bad = fill_mask | nan_mask
 343      n_bad = int(bad.sum())
 344      n_valid = total - n_bad
 345      pct = 100.0 * n_bad / total if total > 0 else 0.0
 346      return {"name": name, "total": total, "valid": n_valid,
 347              "fill_nan": n_bad, "pct": pct}

 415  def _plot_on_ax(ax, data, lons, lats, cmap, vmin, vmax, is_cartopy):
 416      """Plot data on an axis -- tripcolor for 1D, pcolormesh for 2D."""
 417      transform = ccrs.PlateCarree() if is_cartopy else None
 418      if data.ndim == 2:
 419          kw = dict(transform=transform) if is_cartopy else {}
 420          im = ax.pcolormesh(lons, lats, data, cmap=cmap,
 421                             vmin=vmin, vmax=vmax, shading="auto", **kw)
 422      else:
 423          # Unstructured: use tripcolor for filled triangulation.
 424          # Subsample if > 50k points to keep triangulation fast.
 425          # Mask triangles that cross the antimeridian (lon span > 180).
 426          from matplotlib.tri import Triangulation
 427          lons_fix = _fix_lon(lons)
 428          kw = dict(transform=ccrs.PlateCarree()) if is_cartopy else {}
 429          max_pts = 50000
 430          if data.size > max_pts:
 431              idx = np.random.default_rng(42).choice(data.size, max_pts, replace=False)
 432              lons_sub, lats_sub, data_sub = lons_fix[idx], lats[idx], data[idx]
 433          else:
 434              lons_sub, lats_sub, data_sub = lons_fix, lats, data
 435          try:
 436              tri = Triangulation(lons_sub, lats_sub)
 437              # Mask triangles spanning the antimeridian
 438              tri_lons = lons_sub[tri.triangles]
 439              lon_span = tri_lons.max(axis=1) - tri_lons.min(axis=1)
 440              tri.set_mask(lon_span > 180)
 441              im = ax.tripcolor(tri, data_sub, cmap=cmap,
 442                                vmin=vmin, vmax=vmax, **kw)
 443          except Exception:
 444              # Fallback to scatter if tripcolor fails
 445              im = ax.scatter(lons_sub, lats_sub, c=data_sub, s=0.5, cmap=cmap,
 446                              vmin=vmin, vmax=vmax, **kw)
 447      return im

 645  def depth_latitude_section(ds, base_varname, n_levels, title, outdir, fname,
 646                             cmap="RdBu_r", vmin=None, vmax=None, units=""):
 647      """Plot a depth-latitude cross-section from per-level remapped variables.
 648  
 649      For each level k (0 to n_levels-1) reads {base_varname}_{k}, extracts the
 650      last timestep, computes zonal mean, and stacks into a (n_levels, nlat) array.
 651      Depth midpoints are computed from DEPTH_BOUNDS.
 652      """
 653      if not HAS_MPL:
 654          return None
 655  
 656      lat = get_var(ds, "lat")
 657      if lat is None:
 658          return None
 659  
 660      data = []
 661      for k in range(n_levels):
 662          vname = f"{base_varname}_{k}"
 663          arr = get_var(ds, vname)
 664          zm = _compute_zonal_mean(arr)
 665          if zm is None:
 666              return None
 667          data.append(zm)
 668  
 669      data = np.array(data)  # (n_levels, nlat)
 670      n_bounds = min(n_levels + 1, len(DEPTH_BOUNDS))
 671      depth_mid = np.array([(DEPTH_BOUNDS[k] + DEPTH_BOUNDS[k + 1]) / 2.0
 672                            for k in range(n_bounds - 1)])
 673      # Trim data to match available depth bounds
 674      n_plot = min(len(depth_mid), data.shape[0])
 675      depth_mid = depth_mid[:n_plot]
 676      data = data[:n_plot, :]
 677  
 678      fig, ax = plt.subplots(figsize=(10, 5))
 679      im = ax.pcolormesh(lat, depth_mid, data, cmap=cmap,
 680                         vmin=vmin, vmax=vmax, shading="auto")
 681      ax.invert_yaxis()
 682      ax.set_xlabel("Latitude")
 683      ax.set_ylabel("Depth (m)")
 684      ax.set_title(title, fontsize=11)
 685      plt.colorbar(im, ax=ax, label=units)
 686      plt.tight_layout()
 687      return savefig(fig, outdir, fname)

 740  def check_remapped_grid(ds, label):
 741      """Verify that a remapped dataset has the expected lat-lon grid.
 742  
 743      Returns a list of issues (empty if OK).
 744      """
 745      issues = []
 746      dims = get_dims(ds)
 747  
 748      # Check for lon/lat dimensions
 749      has_lon = "lon" in dims
 750      has_lat = "lat" in dims
 751      if not has_lon:
 752          issues.append(f"  {label}: missing 'lon' dimension")
 753      if not has_lat:
 754          issues.append(f"  {label}: missing 'lat' dimension")
 755      if has_lon and dims["lon"] != EXPECTED_NLON:
 756          issues.append(f"  {label}: lon={dims['lon']}, expected {EXPECTED_NLON}")
 757      if has_lat and dims["lat"] != EXPECTED_NLAT:
 758          issues.append(f"  {label}: lat={dims['lat']}, expected {EXPECTED_NLAT}")
 759  
 760      # Check for Time dimension
 761      has_time = "Time" in dims or "time" in dims
 762      if not has_time:
 763          issues.append(f"  {label}: missing 'Time' dimension")
 764  
 765      if not issues:
 766          nlon = dims.get("lon", "?")
 767          nlat = dims.get("lat", "?")
 768          print(f"  {label}: lat-lon grid OK ({nlon}x{nlat})")
 769  
 770      return issues

1366  def read_timing_summary(rundir):
1367      """Read model_timing_stats if present and return summary lines."""
1368      timing_path = os.path.join(rundir, "timing", "model_timing_stats")
1369      if not os.path.exists(timing_path):
1370          return None
1371  
1372      try:
1373          with open(timing_path, "r") as fh:
1374              lines = fh.readlines()
1375      except Exception:
1376          return None
1377  
1378      if not lines:
1379          return None
1380  
1381      # Extract key timing info: look for overall model time and component times
1382      summary = []
1383      for line in lines:
1384          stripped = line.strip()
1385          if not stripped or stripped.startswith("#"):
1386              continue
1387          # Keep header lines and lines with timing data
1388          # Typical format: component-name  seconds  seconds  seconds
1389          if any(kw in stripped.lower() for kw in [
1390              "total", "atm", "lnd", "ocn", "ice", "cpl", "rof", "glc", "wav",
1391              "init", "run", "final", "overall", "model",
1392          ]):
1393              summary.append(stripped)
1394          elif len(summary) == 0:
1395              # Keep initial header/description lines
1396              summary.append(stripped)
1397  
1398      return summary[:40] if summary else None

1742      """Check which ACE-required MPAS variables are present in the FME output.
1743  
1744      Scans all MPAS FME output files (depth coarsening, derived, sea ice).
1745      Atmospheric forcings are marked as 'from EAM' and not checked.
1746      Returns list of (ace_name, fme_name, category_str, found_bool, source) tuples.

1748      # Collect all variable names across MPAS output files
1749      all_vars = set()
1750      patterns = [
1751          "*.mpaso.hist.am.fmeDepthCoarsening.*.remapped.nc",
1752          "*.mpaso.hist.am.fmeDerivedFields.*.remapped.nc",

1757          "*.mpassi.hist.am.fmeSeaiceDerivedFields.*.nc",
1758      ]
1759      for pattern in patterns:
1760          files = find_files(rundir, pattern)
1761          for f in files[:1]:  # only need first file per pattern
1762              ds = safe_open(f)
1763              all_vars.update(get_varnames(ds))
1764              close_ds(ds)

1766      forcing_set = set(ACE_MPAS_FORCING_NAMES)
1767      in_set = set(ACE_MPAS_IN_NAMES)
1768      out_set = set(ACE_MPAS_OUT_NAMES)
1769  
1770      coverage = []
1771      seen = set()
1772      for ace_name in ACE_MPAS_FORCING_NAMES + ACE_MPAS_IN_NAMES + ACE_MPAS_OUT_NAMES:
1773          if ace_name in seen:
1774              continue
1775          seen.add(ace_name)
1776          fme_name = mpas_ace_to_fme(ace_name)
1777          cats = []
1778          if ace_name in forcing_set:
1779              cats.append("forcing")
1780          if ace_name in in_set:
1781              cats.append("in")
1782          if ace_name in out_set:
1783              cats.append("out")
1784          category = ", ".join(cats)
1785  
1786          is_atm = ace_name in ACE_MPAS_ATM_FORCINGS
1787          if is_atm:
1788              source = "EAM"
1789              found = True  # don't check MPAS files for atmospheric vars
1790          else:
1791              source = "MPAS"
1792              found = fme_name in all_vars
1793  
1794          coverage.append((ace_name, fme_name, category, found, source))
