 301  def find_files(rundir, pattern, exclude=None):
 302      """Find files matching glob pattern, optionally excluding a substring.
 303  
 304      Parameters
 305      ----------
 306      rundir : str
 307          Directory to search in.
 308      pattern : str
 309          Glob pattern relative to rundir.
 310      exclude : str or None
 311          If set, exclude any file whose basename contains this substring.
 312  
 313      Returns
 314      -------
 315      list of str
 316          Sorted list of matching file paths.
 317      """
 318      hits = sorted(glob.glob(os.path.join(rundir, pattern)))
 319      # Always exclude .base restart files
 320      hits = [f for f in hits if not f.endswith('.base')]
 321      if exclude:
 322          hits = [f for f in hits if exclude not in os.path.basename(f)]
 323      return hits

 377  def valid_data(arr, fill=1e18):
 378      """Return non-fill, finite values.
 379  
 380      The threshold is 1e18 -- well below SHR_FILL_VALUE (1e20) used by the
 381      FME apply_masked path (gotcha #3, #5, #17 in AGENTS.md), and well above
 382      every geophysical magnitude we output. Notably oceanHeatContent
 383      legitimately reaches ~6e11 J/m^2 in deep equatorial Pacific
 384      (T*h*rho*cp = 30*5000*1025*4000); a tighter threshold mis-classifies
 385      physical OHC as fill. The post-overhaul apply_masked normalizes by
 386      valid_frac and emits exactly SHR_FILL_VALUE (no intermediate
 387      interpolated artifacts), so 1e18 is safe.
 388      """
 389      if arr is None:
 390          return None
 391      flat = arr.ravel().astype(float)
 392      mask = (np.abs(flat) < fill) & np.isfinite(flat)
 393      return flat[mask] if mask.any() else None

 396  def fill_nan_report(arr, name):
 397      """Return a dict with fill/NaN fraction stats for a variable array."""
 398      if arr is None:
 399          return {"name": name, "total": 0, "valid": 0, "fill_nan": 0, "pct": 100.0}
 400      flat = arr.ravel().astype(float)
 401      total = flat.size
 402      fill_mask = np.abs(flat) >= 1e18
 403      nan_mask = ~np.isfinite(flat)
 404      bad = fill_mask | nan_mask
 405      n_bad = int(bad.sum())
 406      n_valid = total - n_bad
 407      pct = 100.0 * n_bad / total if total > 0 else 0.0
 408      return {"name": name, "total": total, "valid": n_valid,
 409              "fill_nan": n_bad, "pct": pct}

 455  def _extract_field(arr, mode):
 456      """Extract a 1D (nCells) field from a multi-dimensional array.
 457  
 458      mode: 'last2d' = last timestep of (Time, nCells)
 459            'surface' = last timestep, surface level of 3D variable.
 460                Detects dimension ordering by size: nCells >> nVertLevels.
 461      """
 462      if arr is None:
 463          return None
 464      if mode == "surface":
 465          if arr.ndim == 3:
 466              if arr.shape[1] < arr.shape[2]:
 467                  return arr[-1, 0, :]   # (Time, nLevels, nCells)
 468              else:
 469                  return arr[-1, :, 0]   # (Time, nCells, nLevels)
 470          elif arr.ndim == 2:
 471              if arr.shape[0] < arr.shape[1]:
 472                  return arr[0, :]       # (nLevels, nCells)
 473              else:
 474                  return arr[:, 0]       # (nCells, nLevels)
 475      else:  # last2d
 476          if arr.ndim == 2:      # (Time, nCells)
 477              return arr[-1, :]
 478          elif arr.ndim == 1:    # (nCells)
 479              return arr[:]
 480      return None

 483  def _diff_stats(leg, fme, fill_thresh=1e18):
 484      """Compute difference statistics between two 1D fields on the same grid.
 485  
 486      Returns dict with n_valid, legacy_mean, fme_mean, diff_mean, diff_rms,
 487      diff_max_abs, correlation -- or None if no valid overlap.
 488      """
 489      if leg is None or fme is None:
 490          return None
 491      a = leg.ravel().astype(float)
 492      b = fme.ravel().astype(float)
 493      if a.size != b.size:
 494          return None
 495      ok = ((np.abs(a) < fill_thresh) & np.isfinite(a) &
 496            (np.abs(b) < fill_thresh) & np.isfinite(b))
 497      if ok.sum() == 0:
 498          return None
 499      a, b = a[ok], b[ok]
 500      d = b - a
 501      s = {
 502          "n_valid": int(ok.sum()),
 503          "legacy_mean": float(a.mean()),
 504          "fme_mean": float(b.mean()),
 505          "diff_mean": float(d.mean()),
 506          "diff_rms": float(np.sqrt((d ** 2).mean())),
 507          "diff_max_abs": float(np.abs(d).max()),
 508      }
 509      if a.std() > 0 and b.std() > 0:
 510          s["correlation"] = float(np.corrcoef(a, b)[0, 1])
 511      else:
 512          s["correlation"] = float("nan")
 513      return s

 520  def offline_depth_coarsen_avg(field, coord_iface, nlev_active, depth_bounds,
 521                                fillval=1e34):
 522      """Overlap-weighted depth coarsening for a single column.
 523  
 524      Exact Python translation of shr_vcoarsen_avg (shr_vcoarsen_mod.F90).
 525  
 526      Parameters
 527      ----------
 528      field : array, shape (nlev,)
 529          Full-resolution field values for one column.
 530      coord_iface : array, shape (nlev+1,)
 531          Depth interfaces (metres), built from restingThickness.
 532      nlev_active : int
 533          Number of active levels (maxLevelCell).
 534      depth_bounds : array, shape (n_out+1,)
 535          Target coarsened layer boundaries (metres).
 536      fillval : float
 537          Fill value for empty layers.
 538  
 539      Returns
 540      -------
 541      field_out : array, shape (n_out,)
 542      """
 543      n_out = len(depth_bounds) - 1
 544      field_out = np.full(n_out, fillval)
 545  
 546      for kout in range(n_out):
 547          b_lo = depth_bounds[kout]
 548          b_hi = depth_bounds[kout + 1]
 549          numerator = 0.0
 550          denominator = 0.0
 551  
 552          for k in range(nlev_active):
 553              c_lo = coord_iface[k]
 554              c_hi = coord_iface[k + 1]
 555              overlap = max(0.0, min(c_hi, b_hi) - max(c_lo, b_lo))
 556              if overlap > 0.0:
 557                  fval = float(field[k])
 558                  if not np.isfinite(fval) or fval == fillval:
 559                      continue
 560                  numerator += fval * overlap
 561                  denominator += overlap
 562  
 563          if denominator > 0.0:
 564              field_out[kout] = numerator / denominator
 565  
 566      return field_out
 567

 569  def offline_depth_coarsen_thickness(resting_thick, nlev_active, depth_bounds):
 570      """Compute coarsened layer thickness from restingThickness.
 571  
 572      Exact Python translation of accumulate_thickness in
 573      mpas_ocn_fme_depth_coarsening.F.
 574  
 575      Parameters
 576      ----------
 577      resting_thick : array, shape (nlev,)
 578          Resting layer thicknesses for one column.
 579      nlev_active : int
 580          Number of active levels (maxLevelCell).
 581      depth_bounds : array, shape (n_out+1,)
 582          Target coarsened layer boundaries (metres).
 583  
 584      Returns
 585      -------
 586      thick_out : array, shape (n_out,)
 587          Coarsened thickness (sum of overlaps). Zero where no overlap.
 588      """
 589      n_out = len(depth_bounds) - 1
 590      thick_out = np.zeros(n_out)
 591      depth_top = 0.0
 592  
 593      for k in range(nlev_active):
 594          thickness = float(resting_thick[k])
 595          depth_bot = depth_top + thickness
 596          for kout in range(n_out):
 597              overlap_top = max(depth_top, depth_bounds[kout])
 598              overlap_bot = min(depth_bot, depth_bounds[kout + 1])
 599              overlap = max(0.0, overlap_bot - overlap_top)
 600              thick_out[kout] += overlap
 601          depth_top = depth_bot
 602  
 603      return thick_out
 604

 606  def offline_depth_coarsen_column(field_2d, resting_thick_2d, max_level_cell,
 607                                   depth_bounds, fillval=1e34):
 608      """Offline depth coarsening for all columns (vectorized outer loop).
 609  
 610      Parameters
 611      ----------
 612      field_2d : array, shape (nCells, nVertLevels)
 613          Full-resolution field.
 614      resting_thick_2d : array, shape (nCells, nVertLevels)
 615          Resting layer thicknesses.
 616      max_level_cell : array, shape (nCells,), int
 617          Number of active levels per cell.
 618      depth_bounds : array, shape (n_out+1,)
 619          Target layer boundaries (metres).
 620  
 621      Returns
 622      -------
 623      coarsened : array, shape (nCells, n_out)
 624          Coarsened field values.
 625      thick_coarsened : array, shape (nCells, n_out)
 626          Coarsened layer thickness.
 627      """
 628      nCells = field_2d.shape[0]
 629      nlev = field_2d.shape[1]
 630      n_out = len(depth_bounds) - 1
 631      coarsened = np.full((nCells, n_out), fillval)
 632      thick_coarsened = np.full((nCells, n_out), fillval)
 633  
 634      depth_bounds_f = np.asarray(depth_bounds, dtype=float)
 635  
 636      for iCell in range(nCells):
 637          ml = int(max_level_cell[iCell])
 638          if ml <= 0:
 639              continue
 640  
 641          # Build coord_iface from restingThickness
 642          coord_iface = np.zeros(nlev + 1)
 643          for k in range(ml):
 644              coord_iface[k + 1] = coord_iface[k] + resting_thick_2d[iCell, k]
 645  
 646          coarsened[iCell, :] = offline_depth_coarsen_avg(
 647              field_2d[iCell, :], coord_iface, ml, depth_bounds_f, fillval)
 648  
 649          tc = offline_depth_coarsen_thickness(
 650              resting_thick_2d[iCell, :], ml, depth_bounds_f)
 651          for kout in range(n_out):
 652              thick_coarsened[iCell, kout] = tc[kout] if tc[kout] > 0 else fillval
 653  
 654      return coarsened, thick_coarsened

 657  def check_offline_depth_coarsening(rundir, outdir, verbose, legacy_rundir=None):
 658      """Cross-verify online FME depth coarsening against offline Python recomputation.
 659  
 660      Reads full-resolution temperature + restingThickness from available native
 661      files, recomputes depth-overlap-weighted coarsening in Python, and compares
 662      against the online FME temperatureCoarsened output.
 663  
 664      Returns (issues, plots, fill_reports).
 665      """
 666      print("\n=== Offline Depth Coarsening Cross-Check ===")
 667      issues = []
 668      plots = []
 669      fill_reports = []
 670      fillval = 1e34
 671  
 672      # --- 1. Find online FME depth-coarsened output (native) ---
 673      fme_files = find_files(rundir,
 674                             "*.mpaso.hist.am.fmeDepthCoarsening.*.nc",
 675                             exclude=".remapped.")
 676      if not fme_files:
 677          print("  SKIP: no native fmeDepthCoarsening files")
 678          return issues, plots, fill_reports
 679  
 680      ds_fme = safe_open(fme_files[-1])
 681      if has_time_zero(ds_fme):
 682          close_ds(ds_fme)
 683          print("  SKIP: time dimension empty")
 684          return issues, plots, fill_reports
 685  
 686      temp_coarsened = get_var(ds_fme, "temperatureCoarsened")
 687      if temp_coarsened is None:
 688          close_ds(ds_fme)
 689          print("  SKIP: no temperatureCoarsened in FME output")
 690          return issues, plots, fill_reports
 691  
 692      # Detect ACTIVE coarsened layer count by inspecting the data, not the
 693      # Registry dimension. The XML allocates nFmeDepthLevels=25 (max) but the
 694      # namelist `config_AM_fmeDepthCoarsening_depth_bounds` selects fewer
 695      # layers (production ships 19); the unused trailing slots are fill.
 696      # We count non-fill slabs in temperatureCoarsened to recover the active
 697      # count, then match DEPTH_BOUNDS against active+1.
 698      if temp_coarsened.ndim == 3:
 699          # Shape is (Time, nCells, nLevels) or (Time, nLevels, nCells); the
 700          # depth axis is the smaller of the two non-time axes.
 701          s1, s2 = temp_coarsened.shape[1], temp_coarsened.shape[2]
 702          depth_axis_len = min(s1, s2)
 703          depth_axis = 1 if temp_coarsened.shape[1] < temp_coarsened.shape[2] else 2
 704      elif temp_coarsened.ndim == 2:
 705          depth_axis_len = min(temp_coarsened.shape)
 706          depth_axis = 0 if temp_coarsened.shape[0] < temp_coarsened.shape[1] else 1
 707      else:
 708          close_ds(ds_fme)
 709          print("  SKIP: temperatureCoarsened has unexpected ndim")
 710          return issues, plots, fill_reports
 711  
 712      # An active layer is one that has at least one finite, non-fill cell.
 713      active_layers = 0
 714      for k in range(depth_axis_len):
 715          sl = np.take(temp_coarsened, k, axis=depth_axis).astype(float)
 716          if (np.isfinite(sl) & (np.abs(sl) < 1e18)).any():
 717              active_layers += 1
 718          else:
 719              break  # active layers are contiguous from k=0
 720      n_coarsen = active_layers
 721  
 722      depth_bounds = np.array(DEPTH_BOUNDS, dtype=float)
 723      if len(depth_bounds) != n_coarsen + 1:
 724          print(f"  SKIP: DEPTH_BOUNDS has {len(depth_bounds)} entries "
 725                f"but file has {n_coarsen} active coarsened layer(s) "
 726                f"(allocated dim: {depth_axis_len}). Update DEPTH_BOUNDS in "
 727                f"verify_mpas.py to match config_AM_fmeDepthCoarsening_depth_bounds.")
 728          close_ds(ds_fme)
 729          return issues, plots, fill_reports
 730  
 731      # Get restingThickness + maxLevelCell from FME file or shared mesh
 732      resting = get_var(ds_fme, "restingThickness")
 733      max_lev = get_var(ds_fme, "maxLevelCell")
 734      close_ds(ds_fme)
 735  
 736      # --- 2. Find full-resolution temperature ---
 737      # Try: FME rundir native output, legacy rundir, restart files
 738      raw_temp = None
 739      raw_resting = resting
 740      raw_maxlev = max_lev
 741      raw_source = None
 742  
 743      search_dirs = [rundir]
 744      if legacy_rundir:
 745          search_dirs.append(legacy_rundir)
 746  
 747      for sdir in search_dirs:
 748          # Try timeSeriesStatsCustom (time-averaged, (Time, nCells, nVertLevels))
 749          for pat in ["*.mpaso.hist.am.timeSeriesStatsCustom.*.nc",
 750                      "*.mpaso.hist.am.timeSeriesStatsDaily.*.nc"]:
 751              for fpath in find_files(sdir, pat, exclude=".remapped."):
 752                  ds = safe_open(fpath)
 753                  # Try multiple variable names
 754                  for tname in ["timeCustom_avg_activeTracers_temperature",
 755                                "timeDaily_avg_activeTracers_temperature",
 756                                "temperature"]:
 757                      arr = get_var(ds, tname)
 758                      if arr is not None and arr.ndim >= 2:
 759                          raw_temp = arr
 760                          raw_source = os.path.basename(fpath)
 761                          # Also grab mesh vars if not yet found
 762                          if raw_resting is None:
 763                              raw_resting = get_var(ds, "restingThickness")
 764                          if raw_maxlev is None:
 765                              raw_maxlev = get_var(ds, "maxLevelCell")
 766                          break
 767                  if raw_temp is not None:
 768                      close_ds(ds)
 769                      break
 770                  close_ds(ds)
 771              if raw_temp is not None:
 772                  break
 773          if raw_temp is not None:
 774              break
 775  
 776      if raw_temp is None:
 777          print("  SKIP: no full-resolution temperature found "
 778                "(need legacy or restart files)")
 779          return issues, plots, fill_reports
 780      if raw_resting is None or raw_maxlev is None:
 781          print("  SKIP: missing restingThickness or maxLevelCell")
 782          return issues, plots, fill_reports
 783  
 784      print(f"  Raw temperature source: {raw_source}")
 785  
 786      # --- 3. Extract last timestep and orient dimensions ---
 787      # raw_temp could be (Time, nCells, nVertLevels) or (Time, nVertLevels, nCells)
 788      if raw_temp.ndim == 3:
 789          raw_temp = raw_temp[-1]  # last timestep -> (dim1, dim2)
 790      # Detect: nCells >> nVertLevels
 791      if raw_temp.ndim == 2:
 792          if raw_temp.shape[0] < raw_temp.shape[1]:
 793              raw_temp = raw_temp.T  # -> (nCells, nVertLevels)
 794      nCells_raw, nVert_raw = raw_temp.shape
 795  
 796      # Same for restingThickness
 797      if raw_resting.ndim == 3:
 798          raw_resting = raw_resting[-1]
 799      if raw_resting.ndim == 2:
 800          if raw_resting.shape[0] < raw_resting.shape[1]:
 801              raw_resting = raw_resting.T
 802      nCells_rt = raw_resting.shape[0]
 803  
 804      # FME online output: last timestep
 805      ds_fme2 = safe_open(fme_files[-1])
 806      temp_coarsened = get_var(ds_fme2, "temperatureCoarsened")
 807      thick_coarsened_online = get_var(ds_fme2, "layerThicknessCoarsened")
 808      close_ds(ds_fme2)
 809  
 810      if temp_coarsened.ndim == 3:
 811          temp_coarsened = temp_coarsened[-1]
 812      if temp_coarsened.ndim == 2:
 813          if temp_coarsened.shape[0] < temp_coarsened.shape[1]:
 814              temp_coarsened = temp_coarsened.T  # -> (nCells, nCoarsen)
 815  
 816      if thick_coarsened_online is not None:
 817          if thick_coarsened_online.ndim == 3:
 818              thick_coarsened_online = thick_coarsened_online[-1]
 819          if thick_coarsened_online.ndim == 2:
 820              if thick_coarsened_online.shape[0] < thick_coarsened_online.shape[1]:
 821                  thick_coarsened_online = thick_coarsened_online.T
 822  
 823      # Check dimension compatibility
 824      nCells_fme = temp_coarsened.shape[0]
 825      if nCells_raw != nCells_fme:
 826          print(f"  SKIP: nCells mismatch: raw={nCells_raw} vs FME={nCells_fme}")
 827          return issues, plots, fill_reports
 828  
 829      # --- 4. Subsample for speed (full recomputation on 465K cells is slow) ---
 830      max_cells = 10000
 831      rng = np.random.default_rng(42)
 832      if nCells_raw > max_cells:
 833          sample_idx = rng.choice(nCells_raw, max_cells, replace=False)
 834          print(f"  Subsampling {max_cells} of {nCells_raw} cells for speed")
 835      else:
 836          sample_idx = np.arange(nCells_raw)
 837  
 838      # --- 5. Recompute offline ---
 839      print(f"  Recomputing depth coarsening for {len(sample_idx)} cells...")
 840      offline_temp, offline_thick = offline_depth_coarsen_column(
 841          raw_temp[sample_idx],
 842          raw_resting[sample_idx],
 843          raw_maxlev[sample_idx],
 844          depth_bounds,
 845          fillval=fillval)
 846  
 847      online_temp = temp_coarsened[sample_idx]
 848      if thick_coarsened_online is not None:
 849          online_thick = thick_coarsened_online[sample_idx]
 850      else:
 851          online_thick = None
 852  
 853      # --- 6. Compare temperature coarsening ---
 854      print("\n  Temperature coarsening (offline vs online):")
 855      temp_issues = 0
 856      for kout in range(n_coarsen):
 857          off = offline_temp[:, kout]
 858          on = online_temp[:, kout]
 859          # Mask fill values
 860          ok = ((np.abs(off) < 1e18) & np.isfinite(off) &
 861                (np.abs(on) < 1e18) & np.isfinite(on))
 862          if ok.sum() == 0:
 863              continue
 864          diff = off[ok].astype(np.float64) - on[ok].astype(np.float64)
 865          rms = float(np.sqrt((diff ** 2).mean()))
 866          max_abs = float(np.abs(diff).max())
 867          mean_d = float(diff.mean())
 868          # Tolerance: should be near-BFB if same input data, but allow for
 869          # time-averaging differences (legacy is time-avg, FME may be instant)
 870          status = "OK" if max_abs < 1.0 else "MISMATCH"
 871          if status == "MISMATCH":
 872              temp_issues += 1
 873              issues.append(f"  temperatureCoarsened layer {kout}: "
 874                            f"max_diff={max_abs:.4g} degC, rms={rms:.4g}")
 875          if verbose or status == "MISMATCH":
 876              print(f"    layer {kout:2d}: mean_diff={mean_d:+.4e}  "
 877                    f"rms={rms:.4e}  max={max_abs:.4e}  "
 878                    f"n={ok.sum()}  {status}")
 879      if temp_issues == 0:
 880          print("    All layers OK (max_diff < 1.0 degC)")
 881  
 882      # --- 7. Compare thickness coarsening ---
 883      if online_thick is not None:
 884          print("\n  Thickness coarsening (offline vs online):")
 885          thick_issues = 0
 886          for kout in range(n_coarsen):
 887              off = offline_thick[:, kout]
 888              on = online_thick[:, kout]
 889              ok = ((np.abs(off) < 1e18) & np.isfinite(off) &
 890                    (np.abs(on) < 1e18) & np.isfinite(on))
 891              if ok.sum() == 0:
 892                  continue
 893              diff = off[ok].astype(np.float64) - on[ok].astype(np.float64)
 894              rms = float(np.sqrt((diff ** 2).mean()))
 895              max_abs = float(np.abs(diff).max())
 896              # Thickness should be BFB (both use restingThickness, no averaging)
 897              status = "OK" if max_abs < 0.01 else "MISMATCH"
 898              if status == "MISMATCH":
 899                  thick_issues += 1
 900                  issues.append(f"  layerThicknessCoarsened layer {kout}: "
 901                                f"max_diff={max_abs:.4g} m, rms={rms:.4g}")
 902              if verbose or status == "MISMATCH":
 903                  print(f"    layer {kout:2d}: rms={rms:.4e}  max={max_abs:.4e}  "
 904                        f"n={ok.sum()}  {status}")
 905          if thick_issues == 0:
 906              print("    All layers OK (BFB within 0.01 m)")
 907  
 908      _report("Offline depth coarsening", issues)
 909      return issues, plots, fill_reports

 934  def _plot_on_ax(ax, data, lons, lats, cmap, vmin, vmax, is_cartopy):
 935      """Plot data on an axis -- tripcolor for 1D, pcolormesh for 2D."""
 936      # Mask fill values so they render as transparent (not clipped to vmax)
 937      data = np.where((np.abs(data) < 1e18) & np.isfinite(data),
 938                      data.astype(float), np.nan)
 939      transform = ccrs.PlateCarree() if is_cartopy else None
 940      if data.ndim == 2:
 941          kw = dict(transform=transform) if is_cartopy else {}
 942          im = ax.pcolormesh(lons, lats, data, cmap=cmap,
 943                             vmin=vmin, vmax=vmax, shading="auto", **kw)
 944      else:
 945          # Unstructured: use tripcolor for filled triangulation.
 946          # Subsample if > 50k points to keep triangulation fast.
 947          # Mask triangles that cross the antimeridian (lon span > 180).
 948          from matplotlib.tri import Triangulation
 949          lons_fix = _fix_lon(lons)
 950          kw = dict(transform=ccrs.PlateCarree()) if is_cartopy else {}
 951          max_pts = 50000
 952          if data.size > max_pts:
 953              idx = np.random.default_rng(42).choice(data.size, max_pts, replace=False)
 954              lons_sub, lats_sub, data_sub = lons_fix[idx], lats[idx], data[idx]
 955          else:
 956              lons_sub, lats_sub, data_sub = lons_fix, lats, data
 957          try:
 958              tri = Triangulation(lons_sub, lats_sub)
 959              # Mask triangles spanning the antimeridian
 960              tri_lons = lons_sub[tri.triangles]
 961              lon_span = tri_lons.max(axis=1) - tri_lons.min(axis=1)
 962              tri.set_mask(lon_span > 180)
 963              im = ax.tripcolor(tri, data_sub, cmap=cmap,
 964                                vmin=vmin, vmax=vmax, **kw)
 965          except Exception:
 966              # Fallback to scatter if tripcolor fails
 967              im = ax.scatter(lons_sub, lats_sub, c=data_sub, s=0.5, cmap=cmap,
 968                              vmin=vmin, vmax=vmax, **kw)
 969      return im

1153  def _compute_zonal_mean(arr, fill_thresh=1e18):
1154      """Compute zonal mean from (lat, lon) array, masking fill values.
1155  
1156      Latitudes that are all-fill (e.g. polar bands fully covered by ice or
1157      land at depth) produce a "Mean of empty slice" RuntimeWarning from
1158      np.nanmean. That's the correct behavior (return NaN), so silence the
1159      warning rather than the result.
1160      """
1161      if arr is None:
1162          return None
1163      if arr.ndim == 3:
1164          data = arr[-1, :, :]
1165      elif arr.ndim == 2:
1166          data = arr
1167      else:
1168          return None
1169      masked = np.where(np.abs(data) < fill_thresh, data, np.nan)
1170      with warnings.catch_warnings():
1171          warnings.simplefilter("ignore", category=RuntimeWarning)
1172          return np.nanmean(masked, axis=1)

1175  def depth_latitude_section(ds, base_varname, n_levels, title, outdir, fname,
1176                             cmap="RdBu_r", vmin=None, vmax=None, units=""):
1177      """Plot a depth-latitude cross-section from per-level remapped variables.
1178  
1179      For each level k (0 to n_levels-1) reads {base_varname}_{k}, extracts the
1180      last timestep, computes zonal mean, and stacks into a (n_levels, nlat) array.
1181      Depth midpoints are computed from DEPTH_BOUNDS.
1182      """
1183      if not HAS_MPL:
1184          return None
1185  
1186      lat = get_var(ds, "lat")
1187      if lat is None:
1188          return None

1190      data = []
1191      for k in range(n_levels):
1192          vname = f"{base_varname}_{k}"
1193          arr = get_var(ds, vname)
1194          zm = _compute_zonal_mean(arr)
1195          if zm is None:
1196              return None
1197          data.append(zm)
1198  
1199      data = np.array(data)  # (n_levels, nlat)
1200      n_bounds = min(n_levels + 1, len(DEPTH_BOUNDS))
1201      depth_mid = np.array([(DEPTH_BOUNDS[k] + DEPTH_BOUNDS[k + 1]) / 2.0
1202                            for k in range(n_bounds - 1)])
1203      # Trim data to match available depth bounds
1204      n_plot = min(len(depth_mid), data.shape[0])

1210                         vmin=vmin, vmax=vmax, shading="auto")

1215      plt.colorbar(im, ax=ax, label=units)
1216      plt.tight_layout()
1217      return savefig(fig, outdir, fname)

1696  def check_remapped_grid(ds, label):
1697      """Verify that a remapped dataset has the expected lat-lon grid.
1698  
1699      Returns a list of issues (empty if OK).
1700      """
1701      issues = []
1702      dims = get_dims(ds)
1703  
1704      # Check for lon/lat dimensions
1705      has_lon = "lon" in dims
1706      has_lat = "lat" in dims
1707      if not has_lon:
1708          issues.append(f"  {label}: missing 'lon' dimension")
1709      if not has_lat:
1710          issues.append(f"  {label}: missing 'lat' dimension")
1711      if has_lon and dims["lon"] != EXPECTED_NLON:
1712          issues.append(f"  {label}: lon={dims['lon']}, expected {EXPECTED_NLON}")
1713      if has_lat and dims["lat"] != EXPECTED_NLAT:
1714          issues.append(f"  {label}: lat={dims['lat']}, expected {EXPECTED_NLAT}")
1715  
1716      # Check for Time dimension
1717      has_time = "Time" in dims or "time" in dims
1718      if not has_time:
1719          issues.append(f"  {label}: missing 'Time' dimension")
1720  
1721      if not issues:
1722          nlon = dims.get("lon", "?")
1723          nlat = dims.get("lat", "?")
1724          print(f"  {label}: lat-lon grid OK ({nlon}x{nlat})")
1725  
1726      return issues

2321  def _area_weighted_mean(data_1d, area_1d, fill_thresh=1e18):
2322      """Area-weighted mean of a 1D field, excluding fill values."""
2323      ok = (np.abs(data_1d) < fill_thresh) & np.isfinite(data_1d) & np.isfinite(area_1d)
2324      if ok.sum() == 0:
2325          return float("nan")
2326      return float(np.sum(data_1d[ok] * area_1d[ok]) / np.sum(area_1d[ok]))

2329  def _cosine_weighted_mean(data_2d, lat_1d, fill_thresh=1e18):
2330      """Cosine-of-latitude-weighted mean of a (lat, lon) field."""
2331      w = np.cos(np.deg2rad(lat_1d))
2332      w2d = np.broadcast_to(w[:, None], data_2d.shape)
2333      ok = (np.abs(data_2d) < fill_thresh) & np.isfinite(data_2d)
2334      if ok.sum() == 0:
2335          return float("nan")
2336      return float(np.sum(data_2d[ok] * w2d[ok]) / np.sum(w2d[ok]))

4208              return None

4211          return (~np.isfinite(sf)) | (np.abs(sf) >= fill_thresh)

4251          bathymetry). For all other variables, use the surface land_mask.
4252          Returns None if no mask is available (skips the check).
4253          """
4254          parts = vname.rsplit('_', 1)
4255          if len(parts) == 2 and parts[1].isdigit():
4256              k = int(parts[1])
4257              if k in bath_masks:
4258                  return bath_masks[k]
4259          return land_mask

4266          "iceAreaTotal", "iceVolumeTotal",
4267          "snowVolumeTotal", "iceThicknessMean",
4268          "surfaceTemperatureMean",
4269          "airStressZonal", "airStressMeridional",
4270          "uVelocityGeoCell", "vVelocityGeoCell",

4279              for vname in sorted(expected_vars.keys()):
4280                  if expected_vars[vname] != stream:
4281                      continue
4282                  arr = get_var(ds, vname)
4283                  if arr is None:
4284                      continue
4285                  expect_fill = _expected_fill_mask(vname)

4293                      else:
