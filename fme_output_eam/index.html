 176  def compute_pint(ps, hyai, hybi):
 177      """Compute interface pressures from hybrid coordinates.
 178  
 179      Parameters
 180      ----------
 181      ps : ndarray, shape (ncol,)
 182          Surface pressure in Pa.
 183      hyai, hybi : ndarray, shape (nlev+1,)
 184          Hybrid A and B coefficients (interface levels, top to bottom).
 185  
 186      Returns
 187      -------
 188      pint : ndarray, shape (ncol, nlev+1)
 189          Interface pressures in Pa.
 190      """
 191      # pint(i, k) = hyai(k) * P0 + hybi(k) * ps(i)
 192      return hyai[np.newaxis, :] * P0 + hybi[np.newaxis, :] * ps[:, np.newaxis]

 195  def compute_pdel(pint):
 196      """Compute layer thickness in Pa from interface pressures.
 197  
 198      Parameters
 199      ----------
 200      pint : ndarray, shape (ncol, nlev+1)
 201  
 202      Returns
 203      -------
 204      pdel : ndarray, shape (ncol, nlev)
 205      """
 206      return pint[:, 1:] - pint[:, :-1]

 209  def offline_vcoarsen_avg(field, pint, pbounds=None):
 210      """Overlap-weighted vertical averaging onto coarsened pressure layers.
 211  
 212      Implements the same algorithm as shr_vcoarsen_avg in shr_vcoarsen_mod.F90.
 213      For each target layer [pb_top, pb_bot], computes the pressure-weighted mean
 214      of all model levels that overlap with it.
 215  
 216      Parameters
 217      ----------
 218      field : ndarray, shape (ncol, nlev)
 219          Full-resolution field values.
 220      pint : ndarray, shape (ncol, nlev+1)
 221          Interface pressures from compute_pint.
 222      pbounds : ndarray, shape (nlayers+1,), optional
 223          Target layer pressure boundaries. Defaults to VCOARSEN_PBOUNDS.
 224  
 225      Returns
 226      -------
 227      coarsened : ndarray, shape (ncol, nlayers)
 228          Coarsened field values.
 229      """
 230      if pbounds is None:
 231          pbounds = VCOARSEN_PBOUNDS
 232      nlayers = len(pbounds) - 1
 233      ncol, nlev = field.shape
 234  
 235      coarsened = np.zeros((ncol, nlayers))
 236      for layer in range(nlayers):
 237          pb_top = pbounds[layer]
 238          pb_bot = pbounds[layer + 1]
 239  
 240          weight_sum = np.zeros(ncol)
 241          for k in range(nlev):
 242              # Overlap between model level k and target layer
 243              overlap_top = np.maximum(pb_top, pint[:, k])
 244              overlap_bot = np.minimum(pb_bot, pint[:, k + 1])
 245              overlap = np.maximum(0.0, overlap_bot - overlap_top)
 246  
 247              # Skip NaN and fill values — only average valid data
 248              fk = field[:, k]
 249              ok = np.isfinite(fk) & (overlap > 0)
 250              coarsened[ok, layer] += fk[ok] * overlap[ok]
 251              weight_sum[ok] += overlap[ok]
 252  
 253          # Normalize by total overlap of valid levels
 254          valid = weight_sum > 0
 255          coarsened[valid, layer] /= weight_sum[valid]
 256          coarsened[~valid, layer] = np.nan
 257  
 258      return coarsened

 261  def offline_column_integrate(field, pdel):
 262      """Column integration: sum(field * pdel / g) over all levels.
 263  
 264      Parameters
 265      ----------
 266      field : ndarray, shape (ncol, nlev)
 267      pdel : ndarray, shape (ncol, nlev)
 268  
 269      Returns
 270      -------
 271      integrated : ndarray, shape (ncol,)
 272      """
 273      return np.sum(field * pdel / GRAVIT, axis=1)

 356  def valid_data(arr, fill=1e10):
 357      """Return non-fill, finite values.
 358  
 359      The threshold is 1e10 because remapped output can contain interpolated
 360      fill artifacts from neighboring land/ocean cells that can reach ~1e14
 361      scale while _FillValue is 1e34.  No geophysical quantity exceeds 1e10.
 362      """
 363      if arr is None:
 364          return None
 365      flat = arr.ravel().astype(float)
 366      mask = (np.abs(flat) < fill) & np.isfinite(flat)
 367      return flat[mask] if mask.any() else None

 370  def fill_nan_report(arr, name):
 371      """Return a dict with fill/NaN fraction stats for a variable array."""
 372      if arr is None:
 373          return {"name": name, "total": 0, "valid": 0, "fill_nan": 0, "pct": 100.0}
 374      flat = arr.ravel().astype(float)
 375      total = flat.size
 376      fill_mask = np.abs(flat) >= 1e10
 377      nan_mask = ~np.isfinite(flat)
 378      bad = fill_mask | nan_mask
 379      n_bad = int(bad.sum())
 380      n_valid = total - n_bad
 381      pct = 100.0 * n_bad / total if total > 0 else 0.0
 382      return {"name": name, "total": total, "valid": n_valid,
 383              "fill_nan": n_bad, "pct": pct}

 451  def _plot_on_ax(ax, data, lons, lats, cmap, vmin, vmax, is_cartopy):
 452      """Plot data on an axis -- tripcolor for 1D, pcolormesh for 2D."""
 453      # Mask fill values so they render as transparent (not clipped to vmax)
 454      data = np.where((np.abs(data) < 1e10) & np.isfinite(data),
 455                      data.astype(float), np.nan)
 456      transform = ccrs.PlateCarree() if is_cartopy else None
 457      if data.ndim == 2:
 458          kw = dict(transform=transform) if is_cartopy else {}
 459          im = ax.pcolormesh(lons, lats, data, cmap=cmap,
 460                             vmin=vmin, vmax=vmax, shading="auto", **kw)
 461      else:
 462          # Unstructured: use tripcolor for filled triangulation.
 463          # Subsample if > 50k points to keep triangulation fast.
 464          # Mask triangles that cross the antimeridian (lon span > 180).
 465          from matplotlib.tri import Triangulation
 466          lons_fix = _fix_lon(lons)
 467          kw = dict(transform=ccrs.PlateCarree()) if is_cartopy else {}
 468          max_pts = 50000
 469          if data.size > max_pts:
 470              idx = np.random.default_rng(42).choice(data.size, max_pts, replace=False)
 471              lons_sub, lats_sub, data_sub = lons_fix[idx], lats[idx], data[idx]
 472          else:
 473              lons_sub, lats_sub, data_sub = lons_fix, lats, data
 474          try:
 475              tri = Triangulation(lons_sub, lats_sub)
 476              # Mask triangles spanning the antimeridian
 477              tri_lons = lons_sub[tri.triangles]
 478              lon_span = tri_lons.max(axis=1) - tri_lons.min(axis=1)
 479              tri.set_mask(lon_span > 180)
 480              im = ax.tripcolor(tri, data_sub, cmap=cmap,
 481                                vmin=vmin, vmax=vmax, **kw)
 482          except Exception:
 483              # Fallback to scatter if tripcolor fails
 484              im = ax.scatter(lons_sub, lats_sub, c=data_sub, s=0.5, cmap=cmap,
 485                              vmin=vmin, vmax=vmax, **kw)
 486      return im

 672      """Compute zonal mean from (lat, lon) array, masking fill values."""
 673      if arr is None:
 674          return None

 676          data = arr[-1, :, :]
 677      elif arr.ndim == 2:
 678          data = arr
 679      else:
 680          return None
 681      masked = np.where(np.abs(data) < fill_thresh, data, np.nan)
 682      return np.nanmean(masked, axis=1)

1553                      print(f"  {fme_base}: SKIP (legacy missing {src})")
1554                      missing_src = True
1555                      break
1556                  src_1t = src_data[tidx]  # (lev, ncol) or (ncol,) from EAM NetCDF
1557                  if src_1t.ndim < 2:
1558                      print(f"  {fme_base}: SKIP ({src} is not 3D)")
1559                      missing_src = True
1560                      break
1561                  # EAM NetCDF stores 3D as (time, lev, ncol); after time slice: (lev, ncol)
1562                  # offline_vcoarsen_avg expects (ncol, lev) — transpose if needed
1563                  if src_1t.shape[0] < src_1t.shape[1]:
1564                      src_1t = src_1t.T  # (lev, ncol) -> (ncol, lev)
1565                  if raw_3d is None:
1566                      raw_3d = src_1t.astype(np.float64)
1567                  else:
1568                      raw_3d = raw_3d + src_1t.astype(np.float64)
1569  
1570              if missing_src or raw_3d is None:
1571                  continue
1572  
1573              # Offline vcoarsen
1574              offline_vc = offline_vcoarsen_avg(raw_3d, pint)  # (ncol, 8)
1575  
1576              # Compare with online vcoarsen from FME output
1577              layer_results = []
1578              max_err_all = 0.0
1579              max_rel_all = 0.0
1580              for k in range(N_VCOARSEN_LAYERS):
1581                  vname = f"{fme_base}_{k}"
1582                  fme_arr = get_var(ds_fme, vname)
1583                  if fme_arr is None:
1584                      layer_results.append(f"  {vname}: MISSING in FME output")
1585                      continue
1586  
1587                  fme_1t = fme_arr[tidx].astype(np.float64).ravel()  # flatten (lat,lon) or (ncol,)
1588                  off_1t = offline_vc[:, k]
1589  
1590                  # Grids may differ (remapped vs native) — compare area-weighted global means
1591                  if fme_1t.size != off_1t.size:
1592                      # Area-weight the FME (lat-lon) data
1593                      fme_lat = get_var(ds_fme, "lat")
1594                      fme_arr_2d = fme_arr[tidx].astype(np.float64)
1595                      if fme_lat is not None and fme_arr_2d.ndim == 2:
1596                          coslat = np.cos(np.deg2rad(fme_lat))
1597                          wt = coslat[:, np.newaxis] * np.ones(fme_arr_2d.shape[1])[np.newaxis, :]
1598                          m = (np.abs(fme_arr_2d) < 1e10) & np.isfinite(fme_arr_2d)
1599                          fme_wm = float(np.average(fme_arr_2d[m], weights=wt[m])) if m.any() else float('nan')
1600                      else:
1601                          fv = valid_data(fme_1t)
1602                          fme_wm = float(fv.mean()) if fv is not None else float('nan')
1603                      off_v = valid_data(off_1t)
1604                      off_wm = float(off_v.mean()) if off_v is not None else float('nan')
1605                      if not np.isnan(fme_wm) and not np.isnan(off_wm):
1606                          rel = abs(fme_wm - off_wm) / max(abs(off_wm), 1e-30)
1607                          max_rel_all = max(max_rel_all, rel)
1608                          max_err_all = max(max_err_all, abs(fme_wm - off_wm))
1609                      continue
1610  
1611                  # Same grid: point-by-point comparison
1612                  valid_mask = (np.abs(fme_1t) < 1e10) & (np.abs(off_1t) < 1e10) & \
1613                               np.isfinite(fme_1t) & np.isfinite(off_1t)
1614  
1615                  if not valid_mask.any():
1616                      layer_results.append(f"  {vname}: no valid data")
1617                      continue
1618  
1619                  diff = np.abs(fme_1t[valid_mask] - off_1t[valid_mask])
1620                  maxdiff = float(diff.max())
1621                  scale = np.maximum(np.abs(off_1t[valid_mask]), 1e-30)
1622                  maxrel = float((diff / scale).max())
1623  
1624                  max_err_all = max(max_err_all, maxdiff)
1625                  max_rel_all = max(max_rel_all, maxrel)
1626  
1627              # Report — thresholds depend on whether grids match.
1628              # For near-zero fields (U, V, TAUX, TAUY), relative difference is
1629              # meaningless because the global mean crosses zero.  Use absolute
1630              # difference instead, benchmarked against a typical scale.
1631              if grids_match:
1632                  bfb_thresh, close_thresh = 1e-10, 1e-5
1633              else:
1634                  bfb_thresh, close_thresh = 1e-5, 0.02  # remap interpolation ~2%
1635  
1636              # Determine a characteristic scale for this field to judge abs diff
1637              off_v = valid_data(offline_vc.ravel())
1638              field_scale = float(np.percentile(np.abs(off_v), 95)) if off_v is not None and off_v.size > 0 else 1.0
1639              # For near-zero-mean fields, use abs diff / field_scale as the metric
1640              if field_scale > 0:
1641                  abs_rel = max_err_all / field_scale
1642              else:
1643                  abs_rel = 0.0
1644  
1645              # Use the more lenient of relative-mean or absolute-relative
1646              effective_metric = min(max_rel_all, abs_rel)
1647  
1648              if effective_metric < bfb_thresh:
1649                  status = "BFB"
1650                  tag = "PASS"
1651              elif effective_metric < close_thresh:
1652                  status = "CLOSE"
1653                  tag = "PASS"
1654              else:
1655                  status = "DIFF"
1656                  tag = "FAIL"
1657                  issues.append(f"  VCOARSEN {fme_base}: max|rel|={max_rel_all:.2e}, "
1658                                f"|diff|/scale={abs_rel:.2e}")
1659  
1660              detail = (f"max|diff|={max_err_all:.2e}, max|rel|={max_rel_all:.2e}, "
1661                        f"|diff|/p95={abs_rel:.2e}")
1662              results.append({"test": "VCOARSEN", "field": fme_base,
1663                              "status": status, "detail": detail})
1664              print(f"  {fme_base}_0..{fme_base}_{N_VCOARSEN_LAYERS-1}: {tag}  {detail}")
1665  
1666              # Generate difference map for last layer (skip if grids differ)
1667              if HAS_MPL and HAS_XR and grids_match:
1668                  last_k = N_VCOARSEN_LAYERS - 1
1669                  vname = f"{fme_base}_{last_k}"
1670                  fme_arr = get_var(ds_fme, vname)
1671                  if fme_arr is not None:
1672                      fme_1t = fme_arr[tidx].astype(np.float64)
1673                      off_1t = offline_vc[:, last_k]
1674                      diff_map = fme_1t - off_1t
1675  
1676                      # Get coordinates
1677                      lon_name = next((d for d in ["lon", "ncol"] if d in get_dims(ds_fme)), None)
1678                      lat_var = get_var(ds_fme, "lat")
1679                      lon_var = get_var(ds_fme, "lon")
1680                      if lat_var is not None and lon_var is not None:
1681                          if lat_var.ndim == 1 and lon_var.ndim == 1:
1682                              lons, lats = np.meshgrid(lon_var, lat_var)
1683                          else:
1684                              lons, lats = lon_var, lat_var
1685                          # Side-by-side: online vs offline
1686                          vmin_f = float(np.nanpercentile(valid_data(fme_1t) or [0], 2))
1687                          vmax_f = float(np.nanpercentile(valid_data(fme_1t) or [0], 98))
1688                          p = side_by_side_comparison(
1689                              fme_1t, lons, lats, f"Online {vname}",
1690                              off_1t, lons, lats, f"Offline vcoarsen({'+'.join(legacy_sources)})",
1691                              f"{fme_base}: Online vs Offline (layer {last_k})",
1692                              xv_outdir, f"xv_vcoarsen_{fme_base}_{last_k}.png",
1693                              cmap="RdYlBu_r", vmin=vmin_f, vmax=vmax_f)
1694                          if p:
1695                              xv_plots.append(p)
1696  
1697      # -- 3. VCOARSEN LINEARITY: STW_k == Q_k + CLDICE_k + ... ----------------
1698      print("\n--- Test 3: Vcoarsen Linearity (STW_k = sum of components) ---")
1699      # If we have offline vcoarsen of individual components (Q, CLDICE, CLDLIQ,
1700      # RAINQM), their sum should equal the online STW_k (since vcoarsen is linear).
1701      if hyai is not None and hybi is not None and leg_ps is not None:
1702          tidx = -1
1703          ps_1t = leg_ps[tidx]
1704          pint = compute_pint(ps_1t, hyai, hybi)
1705  
1706          # Offline vcoarsen each component
1707          components_vc = {}
1708          all_found = True
1709          for comp in ["Q", "CLDICE", "CLDLIQ", "RAINQM"]:
1710              arr = get_var(ds_leg, comp)
1711              if arr is None or arr.ndim < 3:
1712                  all_found = False
1713                  break
1714              comp_1t = arr[tidx].astype(np.float64)
1715              if comp_1t.shape[0] < comp_1t.shape[1]:
1716                  comp_1t = comp_1t.T  # (lev, ncol) -> (ncol, lev)
1717              components_vc[comp] = offline_vcoarsen_avg(comp_1t, pint)
1718  
1719          if all_found:
1720              sum_vc = sum(components_vc.values())  # (ncol, 8)
1721              max_lin_err = 0.0
1722              max_lin_rel = 0.0
1723  
1724              for k in range(N_VCOARSEN_LAYERS):
1725                  stw_k = get_var(ds_fme, f"STW_{k}")
1726                  if stw_k is None:
1727                      continue
1728                  stw_1t = stw_k[tidx].astype(np.float64).ravel()
1729                  sum_1t = sum_vc[:, k]
1730  
1731                  # Different grids: compare global means
1732                  if stw_1t.size != sum_1t.size:
1733                      sv = valid_data(stw_1t)
1734                      ov = valid_data(sum_1t)
1735                      if sv is not None and ov is not None:
1736                          rel = abs(sv.mean() - ov.mean()) / max(abs(ov.mean()), 1e-30)
1737                          max_lin_rel = max(max_lin_rel, rel)
1738                          max_lin_err = max(max_lin_err, abs(sv.mean() - ov.mean()))
1739                      continue
1740  
1741                  valid = (np.abs(stw_1t) < 1e10) & np.isfinite(stw_1t)
1742                  if not valid.any():
1743                      continue
1744  
1745                  diff = np.abs(stw_1t[valid] - sum_1t[valid])
1746                  scale = np.maximum(np.abs(sum_1t[valid]), 1e-30)
1747                  max_lin_err = max(max_lin_err, float(diff.max()))
1748                  max_lin_rel = max(max_lin_rel, float((diff / scale).max()))
1749

1751                  status = "PASS"
1752              else:
1753                  status = "FAIL"
1754                  issues.append(f"  LINEARITY STW: max|rel| = {max_lin_rel:.2e}")
1755  
1756              results.append({"test": "LINEARITY", "field": "STW_k",
1757                              "status": status,
1758                              "detail": f"STW_k vs sum(Q_k+CLDICE_k+CLDLIQ_k+RAINQM_k): "
1759                                        f"max|rel|={max_lin_rel:.2e}"})
1760              print(f"  STW_k linearity: {status}  max|rel|={max_lin_rel:.2e}")
1761          else:
1762              print("  SKIP linearity: component fields not all found in legacy")
