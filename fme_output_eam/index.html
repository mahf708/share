 229  def offline_vcoarsen_avg(field, pint, level_bounds=None):
 230      """dp-weighted vertical averaging by level index ranges.
 231  
 232      Implements the same algorithm as shr_vcoarsen_avg_by_index_cols in
 233      shr_vcoarsen_mod.F90. For each target layer defined by level index
 234      range [level_bounds[k], level_bounds[k+1]), computes the dp-weighted
 235      mean where dp = pint[:,k+1] - pint[:,k].
 236  
 237      Parameters
 238      ----------
 239      field : ndarray, shape (ncol, nlev)
 240          Full-resolution field values.
 241      pint : ndarray, shape (ncol, nlev+1)
 242          Interface pressures from compute_pint.
 243      level_bounds : ndarray, shape (nlayers+1,), optional
 244          0-based level index boundaries. Defaults to VCOARSEN_LEVEL_BOUNDS.
 245  
 246      Returns
 247      -------
 248      coarsened : ndarray, shape (ncol, nlayers)
 249          Coarsened field values.
 250      """
 251      if level_bounds is None:
 252          level_bounds = VCOARSEN_LEVEL_BOUNDS
 253      nlayers = len(level_bounds) - 1
 254      ncol, nlev = field.shape
 255      dp = pint[:, 1:] - pint[:, :-1]  # (ncol, nlev)
 256  
 257      coarsened = np.full((ncol, nlayers), np.nan)
 258      for layer in range(nlayers):
 259          k_lo = level_bounds[layer]      # 0-based, inclusive
 260          k_hi = level_bounds[layer + 1]  # 0-based, exclusive
 261  
 262          if k_lo >= nlev or k_hi <= 0:
 263              continue
 264  
 265          k_lo = max(k_lo, 0)
 266          k_hi = min(k_hi, nlev)
 267  
 268          fslice = field[:, k_lo:k_hi]
 269          dpslice = dp[:, k_lo:k_hi]
 270  
 271          # Mask invalid values
 272          ok = np.isfinite(fslice) & (np.abs(fslice) < 1e10) & (dpslice > 0)
 273  
 274          numerator = np.where(ok, fslice * dpslice, 0.0).sum(axis=1)
 275          denominator = np.where(ok, dpslice, 0.0).sum(axis=1)
 276  
 277          valid = denominator > 0
 278          coarsened[valid, layer] = numerator[valid] / denominator[valid]
 279  
 280      return coarsened

 925      if not os.path.exists(timing_path):

 930              lines = fh.readlines()
 931      except Exception:
 932          return None
 933  
 934      if not lines:
 935          return None
 936  
 937      # Extract key timing info: look for overall model time and component times
 938      summary = []
 939      for line in lines:
 940          stripped = line.strip()
 941          if not stripped or stripped.startswith("#"):
 942              continue
 943          # Keep header lines and lines with timing data

 945          if any(kw in stripped.lower() for kw in [
 946              "total", "atm", "lnd", "ocn", "ice", "cpl", "rof", "glc", "wav",
 947              "init", "run", "final", "overall", "model",
 948          ]):
 949              summary.append(stripped)
 950          elif len(summary) == 0:
 951              # Keep initial header/description lines
 952              summary.append(stripped)
 953  
 954      return summary[:40] if summary else None

1324  def cross_verify(fme_rundir, legacy_rundir, outdir, verbose=False):
1325      """
1326      Compare online FME output against legacy (raw) output.
1327  
1328      Both cases run identical model physics from the same ICs -- the only
1329      difference is what diagnostics get written.  Therefore:
1330  
1331      1. BFB IDENTITY: any field present in both h0 files must match
1332         bit-for-bit (same model state, same history averaging).
1333  
1334      2. ONLINE vs OFFLINE VCOARSEN: the legacy case outputs raw 3D fields
1335         (T, Q, U, V, CLDLIQ, CLDICE, RAINQM at full L80 resolution).
1336         We vcoarsen them offline with the same algorithm / pressure bounds
1337         and compare against the FME case's online T_0..T_7, STW_0..STW_7,
1338         U_0..U_7, V_0..V_7.  These should be BFB identical.
1339  
1340      3. ONLINE vs OFFLINE DERIVED: legacy raw Q + CLDICE + CLDLIQ + RAINQM
1341         should equal FME STW at every level.  After vcoarsening, this tests
1342         derived + vcoarsen together.
1343  
1344      Returns a dict of {test_name: list_of_result_dicts} for HTML rendering.
1345      """
1346      print("\n" + "=" * 70)
1347      print("CROSS-VERIFICATION: fme_output vs fme_legacy_output")
1348      print("=" * 70)
1349  
1350      issues = []
1351      results = []   # list of dicts: {test, field, status, detail}
1352      xv_plots = []  # comparison figures
1353  
1354      xv_outdir = os.path.join(outdir, "fme_output", "cross_verify")
1355      os.makedirs(xv_outdir, exist_ok=True)
1356  
1357      # -- locate h0 files in both rundirs --------------------------------------
1358      fme_h0 = find_files(fme_rundir, "*.eam.h0.*.nc")
1359      leg_h0 = find_files(legacy_rundir, "*.eam.h0.*.nc")
1360  
1361      if not fme_h0 or not leg_h0:
1362          print("  SKIP EAM: h0 files not found in both rundirs")
1363          print(f"    fme: {len(fme_h0)} files,  legacy: {len(leg_h0)} files")
1364          return issues
1365  
1366      ds_fme = safe_open(fme_h0[0])
1367      ds_leg = safe_open(leg_h0[0])
1368  
1369      if has_time_zero(ds_fme) or has_time_zero(ds_leg):
1370          print("  SKIP: time dimension has size 0")
1371          close_ds(ds_fme); close_ds(ds_leg)
1372          return issues
1373  
1374      # -- 1. FIELD IDENTITY ------------------------------------------------------
1375      # Detect grid mismatch (FME may be remapped to lat-lon, legacy on native)
1376      fme_dims = get_dims(ds_fme)
1377      leg_dims = get_dims(ds_leg)
1378      grids_match = ("ncol" in fme_dims) == ("ncol" in leg_dims)
1379  
1380      if grids_match:
1381          print("\n--- Test 1: BFB Identity (shared fields, same grid) ---")
1382      else:
1383          fme_grid = "lat-lon" if "lat" in fme_dims else "native"
1384          leg_grid = "lat-lon" if "lat" in leg_dims else "native"
1385          print(f"\n--- Test 1: Global-Mean Comparison (FME={fme_grid}, Legacy={leg_grid}) ---")
1386  
1387      all_bfb_fields = BFB_INST_FIELDS + BFB_AVG_FIELDS
1388      n_bfb_pass = 0
1389      n_bfb_fail = 0
1390  
1391      for var in all_bfb_fields:
1392          fme_data = get_var(ds_fme, var)
1393          leg_data = get_var(ds_leg, var)
1394          if fme_data is None or leg_data is None:
1395              label = "missing-fme" if fme_data is None else "missing-legacy"
1396              results.append({"test": "BFB", "field": var, "status": "SKIP",
1397                              "detail": f"{label}"})
1398              print(f"  {var}: SKIP ({label})")
1399              continue
1400  
1401          if grids_match and fme_data.shape == leg_data.shape:
1402              # Same grid: point-by-point BFB
1403              diff = np.abs(fme_data.astype(np.float64) - leg_data.astype(np.float64))
1404              maxdiff = float(np.nanmax(diff))
1405              status = "BFB" if maxdiff < 1e-12 else "DIFF"
1406              if status == "BFB":
1407                  n_bfb_pass += 1
1408              else:
1409                  n_bfb_fail += 1
1410                  issues.append(f"  BFB {var}: max|diff| = {maxdiff:.4g}")

1412                              "detail": f"max|diff| = {maxdiff:.2e}"})
1413              tag = "PASS" if status == "BFB" else "FAIL"
1414              print(f"  {var}: {tag}  max|diff| = {maxdiff:.2e}")
1415          else:
1416              # Different grids: compare area-weighted global means (last timestep)
1417              fme_last = fme_data[-1] if fme_data.ndim >= 2 else fme_data

1426                  # cos-lat weighting for regular lat-lon grid
1427                  coslat = np.cos(np.deg2rad(fme_lat))
1428                  weights = coslat[:, np.newaxis] * np.ones(fme_last.shape[1])[np.newaxis, :]
1429                  mask = (np.abs(fme_last) < 1e10) & np.isfinite(fme_last)
1430                  fme_mean = float(np.average(fme_last[mask], weights=weights[mask])) if mask.any() else float('nan')
1431              else:
1432                  fme_v = valid_data(fme_flat)
1433                  fme_mean = float(fme_v.mean()) if fme_v is not None and fme_v.size > 0 else float('nan')
1434  
1435              # Native grid (ne30pg2) is quasi-equal-area, unweighted mean is fine
1436              leg_v = valid_data(leg_flat)
1437              leg_mean = float(leg_v.mean()) if leg_v is not None and leg_v.size > 0 else float('nan')
1438  
1439              if np.isnan(fme_mean) or np.isnan(leg_mean):
1440                  results.append({"test": "GMEAN", "field": var, "status": "SKIP",
1441                                  "detail": "no valid data"})
1442                  print(f"  {var}: SKIP (no valid data)")
1443                  continue
1444  
1445              rel_diff = abs(fme_mean - leg_mean) / max(abs(leg_mean), 1e-30)
1446              abs_diff = abs(fme_mean - leg_mean)
1447              # For near-zero-mean fields (winds, stresses), relative difference
1448              # is meaningless. Use absolute difference against the larger magnitude.
1449              field_mag = max(abs(fme_mean), abs(leg_mean))
1450              # Area-weighted means should agree within remap interpolation error (~2%)
1451              # or within 2% of the field magnitude for near-zero fields
1452              if field_mag > 1e-10:
1453                  effective_rel = abs_diff / field_mag
1454              else:
